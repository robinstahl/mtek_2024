Butterworth-Tiefpassfilter und Anwendung des Filters auf das Signal

In[51]:
def apply_lowpass(signal, cutoff, fs, order=4):
    nyquist = fs * 0.5
    normal_cutoff = cutoff / nyquist
    b, a = butter(order, normal_cutoff, btype='low')
    return filtfilt(b, a, signal)

cutoff = 30
filtered_signal = apply_lowpass(ecg_signal, cutoff, sampling_rate, order=4)
